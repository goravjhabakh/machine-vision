In[8]:
# X and y factor
xFactor = 100
yFactor = 60

# Using opencv
translate = np.float32([[1, 0, xFactor],[0, 1, yFactor]])
translated_img_cv=cv2.warpAffine(img, translate,(cols, rows))

# Without opencv
translated_img_manual=np.zeros_like(img)
for i in range(rows):
    for j in range(cols):
        x=j+xFactor
        y=i+yFactor
        if 0 <= y < rows and 0 <= x < cols:
            translated_img_manual[y,x]=img[i,j]

Scaling

In[25]:
# Scale factors
xScale = 2
yScale = 1.5

# With opencv
center_x, center_y = cols / 2, rows / 2
scale = np.float32([[xScale, 0, center_x * (1 - xScale)], [0, yScale, center_y * (1 - yScale)]])
scaled_img_cv = cv2.warpAffine(img, scale, (cols, rows))

# Without opencv
scaled_img_manual = np.zeros_like(img)
for i in range(rows):
    for j in range(cols):
        x, y = int((j - center_x) / xScale + center_x), int((i - center_y) / yScale + center_y)
        scaled_img_manual[i,j] = img[y,x]

Rotation

In[31]:
angle = 45

# With opencv
center_x, center_y = cols / 2, rows / 2
rotation = cv2.getRotationMatrix2D((center_x, center_y), angle, 1)
rotated_img_cv = cv2.warpAffine(img, rotation, (cols, rows))

# Without opencv
rotated_img_manual = np.zeros_like(img)
theta = np.radians(angle)
for i in range(rows):
    for j in range(cols):
        x = j - center_x
        y = (rows - 1 - i) - center_y 
        new_x = int(center_x + x * np.cos(theta) - y * np.sin(theta))
        new_y = int(center_y + x * np.sin(theta) + y * np.cos(theta))
        if 0 <= new_x < cols and 0 <= new_y < rows:
            rotated_img_manual[rows - 1 - new_y, new_x] = img[i, j]
